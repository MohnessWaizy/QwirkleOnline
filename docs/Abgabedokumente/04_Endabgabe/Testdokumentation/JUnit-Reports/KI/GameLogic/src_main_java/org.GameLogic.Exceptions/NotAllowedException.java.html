package org.GameLogic.Exceptions;

public class NotAllowedException extends Exception {

	private int id;

	public NotAllowedException(String message) {
		this(0, message);
	}

	public NotAllowedException(int id, String message) {
		super(message);
		this.id = id;
	}

	public int getErrorId() {
		return id;
	}

}
